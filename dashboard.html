async function getCurrentUser(){ const { data:{ user }, error } = await sb.auth.getUser(); if(error || !user){ return null; } return user; } async function getProfile(userId){ const { data, error } = await sb .from('profiles') .select('*') .eq('id', userId) .single(); if(error){ throw new Error(error.message); } return data; } async function requireAuth(){ const user = await getCurrentUser(); if(!user){ window.location.href = 'login.html'; return null; } return user; } async function requireRole(expectedRole){ const user = await requireAuth(); if(!user){ return null; } try { const profile = await getProfile(user.id); if(profile.role !== expectedRole){ if(profile.role === 'trader'){ window.location.href = 'trader-dashboard.html'; } else if(profile.role === 'manager'){ window.location.href = 'manager-dashboard.html'; } return null; } return { user, profile }; } catch (err) { window.location.href = 'login.html'; return null; } } async function logout(){ await sb.auth.signOut(); window.location.href = 'login.html'; }
